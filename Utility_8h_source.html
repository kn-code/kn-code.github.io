58///
59/// \brief Computes a superfermion for a given model.
60///
61/// This function computes superfermionic creation/annihilation operators for given models.
62/// The superfermions are superoperators defined as
63/// $$
64/// D^p_{\eta l} \bullet = \frac{1}{\sqrt{2}} ( d_{\eta l} \bullet + p (-\mathbb{1})^n \bullet (-\mathbb{1})^n d_{\eta l} )
65/// $$
66/// where \f$ \bullet \f$ denotes some operator argument.
67///
68/// \param model The model for which the superfermion is computed.
69/// \param p Represents creation or annihilation.
70/// \param eta Either "-" or "+".
71/// \param l Index for internal degrees of freedom (always \f$ \geq 0 \f$ !).
72///
73REALTIMETRANSPORT_EXPORT inline Model::SuperfermionType computeSuperfermion(
74 Keldysh p,
75 Eta eta,
76 int l,
77 const std::unique_ptr<Model>& model)
78{
79 return computeSuperfermion(p, eta, l, model.get());
80}
